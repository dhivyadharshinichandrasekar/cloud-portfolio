# terraform/main.tf - DreamStream Multi-Region Infrastructure

module "vpc_primary" {
  source  = "./modules/vpc"
  
  region              = "us-east-1"
  vpc_cidr            = "10.0.0.0/16"
  availability_zones  = ["us-east-1a", "us-east-1b", "us-east-1c"]
  enable_nat_gateway  = true
  single_nat_gateway  = false
  
  tags = {
    Environment = "production"
    Project     = "dreamstream"
    Region      = "primary"
  }
}

module "eks_cluster" {
  source  = "./modules/eks"
  
  cluster_name    = "dreamstream-primary"
  cluster_version = "1.28"
  vpc_id          = module.vpc_primary.vpc_id
  subnet_ids      = module.vpc_primary.private_subnet_ids
  
  node_groups = {
    streaming = {
      instance_types = ["c5.2xlarge"]
      min_size       = 3
      max_size       = 20
      desired_size   = 5
    }
    general = {
      instance_types = ["m5.xlarge"]
      min_size       = 2
      max_size       = 10
      desired_size   = 3
    }
  }
}

module "rds_primary" {
  source  = "./modules/rds"
  
  identifier        = "dreamstream-db"
  engine            = "aurora-postgresql"

< Key Features />

Multi-Region Deployment

Active-active deployment across 3 AWS regions with Route 53 geolocation routing and automatic failover capabilities.

GitOps with ArgoCD

Declarative GitOps workflow with ArgoCD for automated, auditable, and reversible deployments across all clusters.

Disaster Recovery

RPO < 1 minute with cross-region database replication and RTO < 15 minutes with automated failover procedures.

Infrastructure as Code

100% Terraform-managed infrastructure with modular design, state management in S3, and automated drift detection.

Security First

AWS WAF integration, VPC isolation, secrets management with AWS Secrets Manager, and pod security policies.

Cost Optimization

Spot instances for non-critical workloads, reserved capacity for base load, and automated resource right-sizing.
